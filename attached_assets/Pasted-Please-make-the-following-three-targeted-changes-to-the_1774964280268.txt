```go
// handlePairChannels creates a new channel pairing via Mini App.
// Request body: { "tg_channel_id": -100123456789, "max_chat_id": 987654321, "user_id": 111222333 }
func (b *Bridge) handlePairChannels(w http.ResponseWriter, r *http.Request) {
    if !b.validateMiniAppRequest(w, r) {
        return
    }

    var req struct {
        TgChannelID int64 `json:"tg_channel_id"`
        MaxChatID   int64 `json:"max_chat_id"`
        UserID      int64 `json:"user_id"`
    }
    if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
        http.Error(w, "invalid request body", http.StatusBadRequest)
        return
    }
    if req.TgChannelID == 0 || req.MaxChatID == 0 {
        http.Error(w, "tg_channel_id and max_chat_id are required", http.StatusBadRequest)
        return
    }
    if !b.isUserAllowed(req.UserID) {
        http.Error(w, "forbidden", http.StatusForbidden)
        return
    }

    if err := b.repo.AddCrosspost(req.TgChannelID, req.MaxChatID, req.UserID, req.UserID); err != nil {
        slog.Error("handlePairChannels: AddCrosspost failed", "err", err)
        http.Error(w, "failed to create pairing", http.StatusInternalServerError)
        return
    }

    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(http.StatusCreated)
    json.NewEncoder(w).Encode(map[string]string{"status": "ok"})
}
```

Note: adapt the `AddCrosspost` call signature to match what's already defined in `repository.go` / `sqlite.go` / `postgres.go`.
